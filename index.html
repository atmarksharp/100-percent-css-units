100% CSS Units

Create the 100% width with the numeric units of CSS

100%

What Is This?

This list shows that each numeric units of CSS can be converted to other units with javascript.

CSS Numeric Units

For detail, see CSS Values and Units Module Level 3

  • % (percent) —— percentage
  • px (pixels) —— pixels. 1px is equal to 1/96 inchs.
  • pt (points) —— points. 1pt is equal to 1/72 inchs.
  • em —— font size of the element
  • ex —— small "x"-height of the element's font
  • rem —— font size of the root element
  • pc (picas) —— 1pc is equal to 12pt.
  • in (inches) —— inches.1in is equal to 2.54cm
  • mm (millimeters) —— millimeters
  • cm (centimeters) —— centimeters
  • ch —— width of the "0" (ZERO, U+0030) glyph in the element's font
  • vw —— 1% of viewport width
  • vh —— 1% of viewport height
  • vmin —— 1% of viewport length with minimum-length
  • vmax —— 1% of viewport length with maximum-length

Unit Conversion

  • px in —— 1px = 1/96 in (see the specification)
  • pt in —— 1pt = 1/72 in (see the specification)
  • em px —— get the font-size of the element (see code1)
  • ex, ch px —— create and get the size of a new element that inherits font properties of the element (see code2)
  • rem px —— get the font-size of the root element like <html> (see code3)
  • in px —— 1pc = 12pt, 1pt = 1/72 in and 1in = 96px.
  • pc px —— 1pc = 12pt, and 1pt = 1/72 in.
  • mm, cm px —— 1inch = 2.54cm, 1cm = 10mm and and 1in = 96px.
  • vw, vh, vmin, vmax px —— get the viewport size, and compare sizes when vmin or vmax (see code3)
    code 1: em px
    
        function getFontSizeAsPixels($dom){
          return Number($dom.css('font-size').match(/(\d*(\.\d*)?)px/)[1]);
        }
    code 2: ex, ch px
    
        function getPixelsPerUnit($dom,unit){
          var div = document.createElement("div");
          var st = div.style;
          st.fontSize=$dom.css('font-size');
          st.fontFamily=$dom.css('font-family');
          st.fontWeight=$dom.css('font-weight');
          st.fontVariant=$dom.css('font-variant');
          st.fontStyle=$dom.css('font-style');
          st.lineHeight=$dom.css('line-height');
          st.width='100'+unit
          var body = document.getElementsByTagName("body")[0];
          body.appendChild(div);
          var ppx= document.defaultView.getComputedStyle(div, null).getPropertyValue('width');
          body.removeChild(div);
          return parseFloat(ppx) / 100;
        }

        function getPixelsPerEx($dom){
          return getPixelsPerUnit($dom,'ex');
        }

        function getPixelsPerCh($dom){
          return getPixelsPerUnit($dom,'ch');
        }

    code 4: vw, vh, vmin, vmax px
    
      function getViewportWidth(){
        var div = document.createElement("div");
        div.style.width = '100vw'
        var body = document.getElementsByTagName("body")[0];
        body.appendChild(div);
        var size = document.defaultView.getComputedStyle(div, null).getPropertyValue('width');
        body.removeChild(div);
        return parseFloat(size);
      }

      function getViewportHeight(){
        var div = document.createElement("div");
        div.style.height = '100vh'
        var body = document.getElementsByTagName("body")[0];
        body.appendChild(div);
        var size = document.defaultView.getComputedStyle(div, null).getPropertyValue('height');
        body.removeChild(div);
        return parseFloat(size);
      }

      var viewport_w = getViewportWidth();
      var viewport_h = getViewportHeight();
      var PX_PER_VW = viewport_w * 0.01;
      var PX_PER_VH = viewport_h * 0.01;

      var vw_val = 100;
      var vw_w = '100vw';
      setVal(vw_w,'vw');

      var vh_val = (PX_PER_VW / PX_PER_VH) * 100;
      var vh_w = vh_val + 'vh';
      setVal(vh_w,'vh');

      var vmin_w = (viewport_w < viewport_h)? vw_val+'vmin': vh_val+'vmin';
      setVal(vmin_w,'vmin');

      var vmax_w = (viewport_w >= viewport_h)? vw_val+'vmax': vh_val+'vmax';
      setVal(vmax_w,'vmax');
